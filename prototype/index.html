//Q1.Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype


const parentObj = {
calAge : function(){
console.log(`${this.fname} ${this.lname} ${this.age}`)
}
}
const childObj = Object.create(parentObj);
childObj.fname = "Albert";
childObj.lname = "Mathew";
childObj.age = 23;
childObj.calAge();

output :
Albert Mathew 23

//Q2Write code to explain prototype chaining.


class Person {
constructor() {
this.gender = 'female';
}
printPerson() {}
}
class User extends Person {
constructor() {
super();
this.name = 'Aaisha';
}
printUser() {}
}
var u = new User();
console.log(u);

Output : gender :"female" , name:"Aaisha"
